--------- 找出 Where 欄位
select
r.name owner,
o.name table_name,
c.name column_name,
equality_preds,
equijoin_preds,
nonequijoin_preds,
range_preds,
like_preds,
null_preds,
timestamp
from 
sys.col_usage$ u ,
sys.obj$ o ,
sys.col$ c,
sys.user$ r 
where 
o.obj# = u.obj#
and c.obj# = u.obj#
and c.col# = u.intcol#
and r.name = 'JOHN'
and o.name = 'TEST';
